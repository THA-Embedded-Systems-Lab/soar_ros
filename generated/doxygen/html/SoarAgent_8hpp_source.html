162 void updateWorld()
163 {
164 processOutputLinkChanges();
165 processInput();
166 }
167
168 private:
169 void processOutputLinkChanges()
170 {
171 if (!m_pAgent)
172 {
173 return;
174 }
175
176 int n = m_pAgent->GetNumberCommands();
177 for (int i = 0; i < n; ++i)
178 {
179 sml::Identifier *pId = m_pAgent->GetCommand(i);
180 std::string name = pId->GetCommandName();
181
182 auto it = m_outputs.find(name);
183 if (it == m_outputs.end())
184 {
185 RCLCPP_ERROR(
186 m_node->get_logger(),
187 "[%s] No output mapping for command: %s",
188 m_pAgent->GetAgentName(), name.c_str());
189 continue;
190 }
191 it->second->process_s2r(pId);
192 pId->AddStatusComplete();
193 }
194 }
195
196 void processInput()
197 {
198 if (!m_pAgent)
199 {
200 return;
201 }
202 for (const auto &input : m_inputs)
203 {
204 input->process_r2s();
205 }
206 m_pAgent->Commit();
207 }
208
209 sml::Agent *m_pAgent;
210 rclcpp::Node::SharedPtr m_node;
211
213 std::map<std::string, std::shared_ptr<soar_ros::OutputBase>> m_outputs;
215 std::vector<std::shared_ptr<soar_ros::InputBase>> m_inputs;
216 };
