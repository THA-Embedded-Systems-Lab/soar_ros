71 void send_goal_from_soar(const pGoalMsg & soar_goal)
72 {
73 RCLCPP_DEBUG(m_node->get_logger(), "Sending goal to action server: %s", m_topic.c_str());
74
75 auto options = typename rclcpp_action::Client<ActionT>::SendGoalOptions();
76
77 // Goal response callback
78 options.goal_response_callback = [this](std::shared_ptr<GoalHandle> goal_handle) {
79 if (!goal_handle) {
80 this->template Input<bool>::m_r2sQueue.push(false);
81 } else {
82 this->template Input<bool>::m_r2sQueue.push(true);
83 }
84 };
85
86 // Feedback callback
87 options.feedback_callback = [this](GoalHandle::SharedPtr,
88 const std::shared_ptr<const typename ActionT::Feedback> feedback) {
89 // Create a non-const copy for the queue
90 auto feedback_copy = std::make_shared<typename ActionT::Feedback>(*feedback);
91 this->template Input<pFeedbackMsg>::m_r2sQueue.push(feedback_copy);
92 };
93
94 // Result callback
95 options.result_callback = [this](const wrappedResultMsg & wrapped) {
96 this->template Input<wrappedResultMsg>::m_r2sQueue.push(wrapped);
97 };
98
99 client_ptr_->async_send_goal(*soar_goal, options);
100 }
101
102 // Interface implementation
103 std::string getTopic() override {return m_topic;}
104 sml::Agent * getAgent() override {return m_pAgent;}
105
106protected:
107 using Input<pFeedbackMsg>::parse;
108 using Input<wrappedResultMsg>::parse;

111 virtual pGoalMsg parse(sml::Identifier * id) = 0;
112 virtual void parse(pFeedbackMsg msg) = 0;
113 virtual void parse(wrappedResultMsg msg) = 0;
114
115 void parse(bool msg) override
116 {
117 sml::Identifier * il = this->getAgent()->GetInputLink();
118 sml::Identifier * ros_action_id = il->CreateIdWME(this->m_topic.c_str());
119 ros_action_id->CreateStringWME("status", msg ? "accepted" : "rejected");
120 }
121
122private:
123 sml::Agent * m_pAgent;
124 rclcpp::Node::SharedPtr m_node;
125 std::string m_topic;
126 typename rclcpp_action::Client<ActionT>::SharedPtr client_ptr_;
127
128 std::thread m_action_thread;
129 std::atomic<bool> isRunning;
130
131 // Background thread to process goals
132 void run()
133 {
134 while (isRunning.load()) {
135 // Wait for action server
136 if (!client_ptr_->wait_for_action_server(std::chrono::seconds(1))) {
137 if (!rclcpp::ok()) {
138 RCLCPP_ERROR(m_node->get_logger(), "Interrupted while waiting for action server");
139 isRunning.store(false);
140 break;
141 }
142 continue;
143 }
144
145 // Process any pending goals
146 auto goal = this->template Output<pGoalMsg>::m_s2rQueue.tryPop();
147 if (goal.has_value()) {
148 send_goal_from_soar(goal.value());
149 }
150
151 std::this_thread::sleep_for(std::chrono::milliseconds(10));
152 }
153 }
154};
