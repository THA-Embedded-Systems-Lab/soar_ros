337  void startThread()
338  {
339  // If thread is already running, skip
340  if (isRunning.load() == true) {
341  RCLCPP_WARN(this->get_logger(), "runThread already running");
342  return;
343  } else {
344  isRunning.store(true);
345  }
346 
347  RCLCPP_INFO(this->get_logger(), "Starting runThread");
348  runThread = std::thread(&SoarRunner::run, this);
349  }
350 
353  void stopThread()
354  {
355  if (isRunning.load() == false) {
356  RCLCPP_INFO(this->get_logger(), "Run thread already stopped!");
357  return;
358  }
359 
360  isRunning.store(false);
361  RCLCPP_WARN(this->get_logger(), "Stopping runThread");
362 
363  if (runThread.joinable()) {
364  RCLCPP_INFO(this->get_logger(), "Waiting for run thread to join");
365  runThread.join();
366  }
367  }
